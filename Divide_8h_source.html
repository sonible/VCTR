28template <size_t extent, class SrcAType, class SrcBType>
29class DivideVectors : ExpressionTemplateBase
30{
31public:
32 using value_type = std::common_type_t<ValueType<SrcAType>, ValueType<SrcBType>>;
33
34 VCTR_COMMON_BINARY_VEC_VEC_EXPRESSION_MEMBERS (DivideVectors, srcA, srcB)
35
36 VCTR_FORCEDINLINE constexpr value_type operator[] (size_t i) const
37 {
38 return srcA[i] / srcB[i];
39 }
40
41 VCTR_FORCEDINLINE const value_type* evalNextVectorOpInExpressionChain (value_type* dst) const
42 requires is::suitableForAccelerateRealOrComplexComplexFloatBinaryVectorOp<SrcAType, SrcBType, value_type, detail::dontPreferIfIppAndAccelerateAreAvailable>
43 {
44 Expression::Accelerate::div (srcA.evalNextVectorOpInExpressionChain (dst), srcB.evalNextVectorOpInExpressionChain (dst), dst, size());
45 return dst;
46 }
47
48 VCTR_FORCEDINLINE const value_type* evalNextVectorOpInExpressionChain (value_type* dst) const
49 requires is::suitableForIppRealOrComplexComplexFloatBinaryVectorOp<SrcAType, SrcBType, value_type, detail::preferIfIppAndAccelerateAreAvailable>
50 {
51 Expression::IPP::div (srcA.evalNextVectorOpInExpressionChain (dst), srcB.evalNextVectorOpInExpressionChain (dst), dst, sizeToInt (size()));
52 return dst;
53 }
54
55 //==============================================================================
56 VCTR_FORWARD_PREPARE_SIMD_EVALUATION_BINARY_EXPRESSION_MEMBER_FUNCTIONS (srcA, srcB)
57
58 // AVX Implementation
59 VCTR_FORCEDINLINE VCTR_TARGET ("avx") AVXRegister<value_type> getAVX (size_t i) const
60 requires (archX64 && has::getAVX<SrcAType> && has::getAVX<SrcBType> && Expression::CommonElement::isRealFloat)
61 {
62 return Expression::AVX::div (srcA.getAVX (i), srcB.getAVX (i));
63 }
64
65 // SSE Implementation
66 VCTR_FORCEDINLINE VCTR_TARGET ("sse4.1") SSERegister<value_type> getSSE (size_t i) const
67 requires (archX64 && has::getSSE<SrcAType> && has::getSSE<SrcBType> && Expression::CommonElement::isRealFloat)
68 {
69 return Expression::SSE::div (srcA.getSSE (i), srcB.getSSE (i));
70 }
71};

77{
78public:

81 VCTR_COMMON_BINARY_SINGLE_VEC_EXPRESSION_MEMBERS (DivideSingleByVec, src, single)
82
83 VCTR_FORCEDINLINE constexpr value_type operator[] (size_t i) const
84 {
85 return single / src[i];
86 }
87
88 VCTR_FORCEDINLINE const value_type* evalNextVectorOpInExpressionChain (value_type* dst) const
89 requires is::suitableForIppRealFloatVectorOp<SrcType, value_type> && std::same_as<float, value_type>
90 {
91 Expression::IPP::div (single, src.evalNextVectorOpInExpressionChain (dst), dst, sizeToInt (size()));
92 return dst;
93 }
94
95 //==============================================================================
96 // AVX Implementation
97 void prepareAVXEvaluation() const
98 requires has::prepareAVXEvaluation<SrcType>
99 {
100 src.prepareAVXEvaluation();
101 singleSIMD.avx = Expression::AVX::broadcast (single);
102 }
103
104 VCTR_FORCEDINLINE VCTR_TARGET ("avx") AVXRegister<value_type> getAVX (size_t i) const
105 requires (archX64 && has::getAVX<SrcType> && Expression::allElementTypesSame && Expression::CommonElement::isRealFloat)
106 {
107 return Expression::AVX::div (singleSIMD.avx, src.getAVX (i));
108 }
109
110 // SSE Implementation
111 void prepareSSEEvaluation() const

114 src.prepareSSEEvaluation();
115 singleSIMD.sse = Expression::SSE::broadcast (single);

119 requires (archX64 && has::getSSE<SrcType> && Expression::allElementTypesSame && Expression::CommonElement::isRealFloat)
120 {
121 return Expression::SSE::div (singleSIMD.sse, src.getSSE (i));
122 }

124private:

199template <class SrcAType, class SrcBType>
200requires (is::anyVctrOrExpression<std::remove_cvref_t<SrcAType>> &&
201 is::anyVctrOrExpression<std::remove_cvref_t<SrcBType>>)
202constexpr auto operator/ (SrcAType&& a, SrcBType&& b)
203{
204 assertCommonSize (a, b);
205 constexpr auto extent = getCommonExtent<SrcAType, SrcBType>();
206
207 return expressions::DivideVectors<extent, SrcAType, SrcBType> (std::forward<SrcAType> (a), std::forward<SrcBType> (b));
208}
