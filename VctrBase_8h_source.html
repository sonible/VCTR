281 constexpr void copyFrom (const ElementType* otherData, size_t otherSize)
282 requires is::nonConst<ElementType>
283 {
284 resizeOrAssertSizeMatches (otherSize);
285
286 if constexpr (is::triviallyCopyable<ElementType>)
287 {
288 if (! std::is_constant_evaluated())
289 {
290 std::memcpy (data(), otherData, sizeof (ElementType) * otherSize);
291 return;
292 }
293 }
294
295 std::copy_n (otherData, otherSize, begin());
296 }

316 constexpr void fillLinspace (ElementType start, ElementType stop, bool includeEnd = true)
317 requires is::realNumber<ElementType>
318 {
319 const auto num = size();
320
321 VCTR_ASSERT (num > 0);
322 VCTR_ASSERT (! includeEnd || num != 1); // a num-one range cannot include the end
323
324 const auto increment = (stop - start) / double (num - size_t (includeEnd));
325
326#if VCTR_USE_GCEM
327 [[maybe_unused]] const auto isIntegerIncrement = increment - gcem::floor (increment) == ElementType (0);
328#else
329 [[maybe_unused]] const auto isIntegerIncrement = increment - std::floor (increment) == ElementType (0);
330#endif
331 VCTR_ASSERT (std::is_floating_point_v<ElementType> || isIntegerIncrement);
332
333 auto value = double (start);
334
335 for (int i = 0; i < num; ++i)
336 {
337 storage[i] = ElementType (value);
338 value += increment;
339 };
340 }

696 void shiftLeft (size_t n, bool clearFreeSpaceAfterShiftedRegion)
697 requires std::is_trivially_copyable_v<ElementType>
698 {
699 if (n == size())
700 {
701 if (clearFreeSpaceAfterShiftedRegion)
702 clear (data(), size());
703
704 return;
705 }
706
707 VCTR_ASSERT (n < size());
708
709 auto numElementsToMove = size() - n;
710 memMove (data() + n, data(), numElementsToMove);
711
712 if (clearFreeSpaceAfterShiftedRegion)
713 clear (data() + numElementsToMove, n);
714 }
715
723 void shiftRight (size_t n, bool clearFreeSpaceBeforeShiftedRegion)
724 requires std::is_trivially_copyable_v<ElementType>
725 {
726 if (n == size())
727 {
728 if (clearFreeSpaceBeforeShiftedRegion)
729 clear (data(), size());
730
731 return;
732 }
733
734 VCTR_ASSERT (n < size());
735
736 const auto numElementsToMove = size() - n;
737 memMove (data(), data() + n, numElementsToMove);
738
739 if (clearFreeSpaceBeforeShiftedRegion)
740 clear (data(), n);
741 }
