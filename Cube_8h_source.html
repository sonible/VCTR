28{
29public:

32 VCTR_COMMON_UNARY_EXPRESSION_MEMBERS (Cube, src)
33
34 VCTR_FORCEDINLINE constexpr value_type operator[] (size_t i) const
35 {
36 const auto s = src[i];
37 return s * s * s;
38 }
