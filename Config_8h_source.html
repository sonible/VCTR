158#if VCTR_WINDOWS
159namespace detail
160{
162class X64InstructionSets
163{
164private:
165 struct CPUFeatureFinder
166 {
167 CPUFeatureFinder()
168 {
169 std::array<int, 4> cpui;
170 std::vector<std::array<int, 4>> data, extdata;
171
172 // Calling __cpuid with 0x0 as the function_id argument
173 // gets the number of the highest valid function ID.
174 __cpuid (cpui.data(), 0);
175 auto nIds = cpui[0];
176
177 for (int i = 0; i <= nIds; ++i)
178 {
179 __cpuidex (cpui.data(), i, 0);
180 data.push_back (cpui);
181 }
182
183 // load bitset with flags for function 0x00000001
184 if (nIds >= 1)
185 {
186 f_1_ECX_ = data[1][2];
187 f_1_EDX_ = data[1][3];
188 }
189
190 // load bitset with flags for function 0x00000007
191 if (nIds >= 7)
192 {
193 f_7_EBX_ = data[7][1];
194 f_7_ECX_ = data[7][2];
195 }
196
197 // Calling __cpuid with 0x80000000 as the function_id argument
198 // gets the number of the highest valid extended ID.
199 __cpuid (cpui.data(), 0x80000000);
200 auto nExIds = cpui[0];
201
202 for (int i = 0x80000000; i <= nExIds; ++i)
203 {
204 __cpuidex (cpui.data(), i, 0);
205 extdata.push_back (cpui);
206 }
207
208 // load bitset with flags for function 0x80000001
209 if (nExIds >= 0x80000001)
210 {
211 f_81_ECX_ = extdata[1][2];
212 f_81_EDX_ = extdata[1][3];
213 }
214 };
215
216 std::bitset<32> f_1_ECX_;
217 std::bitset<32> f_1_EDX_;
218 std::bitset<32> f_7_EBX_;
219 std::bitset<32> f_7_ECX_;
220 std::bitset<32> f_81_ECX_;
221 std::bitset<32> f_81_EDX_;
222 };
223
224 inline static const CPUFeatureFinder cpuFeatures;
225
226public:
227 static bool hasFMA() { return cpuFeatures.f_1_ECX_[12]; }
228 static bool hasSSE41() { return cpuFeatures.f_1_ECX_[19]; }
229 static bool hasSSE42() { return cpuFeatures.f_1_ECX_[20]; }
230 static bool hasAVX() { return cpuFeatures.f_1_ECX_[28]; }
231 static bool hasAVX2() { return cpuFeatures.f_7_EBX_[5]; }
232 static bool hasAVX512F() { return cpuFeatures.f_7_EBX_[16]; }
233
234private:
235};
236} // namespace detail

238inline CPUInstructionSet getHighestSupportedCPUInstructionSet()
239{
240 if (detail::X64InstructionSets::hasAVX2())
241 return CPUInstructionSet::avx2;
242
243 if (detail::X64InstructionSets::hasAVX())
244 return CPUInstructionSet::avx;
245
246 if (detail::X64InstructionSets::hasSSE41())
247 return CPUInstructionSet::sse4_1;
248
249 return CPUInstructionSet::fallback;
250}
