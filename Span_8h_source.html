157 template <has::sizeAndData Container>
158 static constexpr auto makeStorageInfo (const Container& container)
159 {
160 constexpr size_t memberAlignment = alignof (StdSpanType);
161
162 if constexpr (is::anyVctr<Container>)
163 {
164 const auto& info = container.getStorageInfo();
165 return StorageInfoWithMemberAlignment<memberAlignment, std::remove_cvref_t<decltype (info)>> (info);
166 }
167 else
168 {
169 return StorageInfoWithMemberAlignment<memberAlignment, StorageInfo<Container>>().init (container.data(), container.size());

190{
191 // If you hit this assertion, you passed a StaticStorageInfo that specifies SIMD aligned
